## Sous-tâches

### 1. Structure du script

```python
# scripts/train_tinyol.py
"""
Entraînement TinyOL en scénario Domain-Incremental sur Dataset 1 (Pump).

Protocole :
  - Tâche 1 : fenêtres t=0..T/3    (pompe saine)
  - Tâche 2 : fenêtres t=T/3..2T/3 (usure naissante)
  - Tâche 3 : fenêtres t=2T/3..T   (pré-panne)
  Après chaque tâche → évaluation sur toutes les tâches vues.

Sorties :
    experiments/exp_003_tinyol_dataset1/results/metrics.json
    experiments/exp_003_tinyol_dataset1/results/cl_accuracy_matrix.png
    experiments/exp_003_tinyol_dataset1/config_snapshot.yaml

Références :
    Ren2021TinyOL, tinyol_spec.md §4 et §6
"""
import argparse
import json
from pathlib import Path

import torch
import yaml


def parse_args() -> argparse.Namespace:
    parser = argparse.ArgumentParser(description="Entraînement TinyOL — exp_003")
    parser.add_argument(
        "--config",
        type=str,
        default="configs/tinyol_config.yaml",
        help="Chemin vers le fichier de configuration YAML",
    )
    return parser.parse_args()


if __name__ == "__main__":
    args = parse_args()
    with open(args.config) as f:
        config = yaml.safe_load(f)
    run_experiment(config)
```

### 2. Chargement et initialisation

```python
from src.data.pump_dataset import PumpMaintenanceDataset, CLStreamSplitter
from src.models.tinyol import TinyOLAutoencoder, OtOHead, TinyOLOnlineTrainer
from src.evaluation.metrics import compute_cl_metrics
from src.evaluation.memory_profiler import measure_inference_latency, measure_ram_update
from src.utils.reproducibility import set_seed


def run_experiment(config: dict) -> None:
    set_seed(config["online"]["seed"])
    output_dir = Path(config["evaluation"]["output_dir"])
    output_dir.mkdir(parents=True, exist_ok=True)

    # --- Données ---
    dataset = PumpMaintenanceDataset(config["data"]["path"])
    features, labels = dataset.extract_features(
        window_size=config["data"]["window_size"],
        step_size=config["data"]["step_size"],
    )
    cl_stream = CLStreamSplitter(
        features, labels,
        n_tasks=config["data"]["n_tasks"],
        strategy=config["data"]["task_strategy"],
    )
    normalizer_path = config["data"]["normalizer_path"]
    cl_stream.load_normalizer(normalizer_path)
    cl_stream.apply_normalizer(cl_stream._normalizer)

    # --- Modèle : backbone gelé + tête OtO fraîche ---
    autoencoder = TinyOLAutoencoder(
        input_dim=config["backbone"]["input_dim"],
        encoder_dims=tuple(config["backbone"]["encoder_dims"]),
        decoder_dims=tuple(config["backbone"]["decoder_dims"]),
    )
    checkpoint_path = Path(config["backbone"]["checkpoint_path"])
    autoencoder.load_state_dict(torch.load(checkpoint_path, map_location="cpu"))
    autoencoder.eval()

    oto_head = OtOHead(input_dim=config["online"]["oto_input_dim"])
    trainer = TinyOLOnlineTrainer(autoencoder, oto_head, config)

    print(f"[exp_003] Backbone chargé depuis {checkpoint_path}")
    print(f"[exp_003] OtO params : {oto_head.n_params()} (attendu : 10)")
    print(f"[exp_003] Stream CL : {config['data']['n_tasks']} tâches")
```

### 3. Boucle CL et évaluation

```python
    # Matrice d'accuracy : acc_matrix[i][j] = acc sur tâche j après entraînement sur tâches 0..i
    n_tasks = config["data"]["n_tasks"]
    acc_matrix: list[list[float]] = []

    for task_id in range(n_tasks):
        print(f"\n--- Tâche {task_id + 1}/{n_tasks} ---")

        # Entraînement online (1 pass, échantillon par échantillon)
        task_features, task_labels = cl_stream.get_task(task_id)
        n_correct, total = 0, 0
        for x_np, y_np in zip(task_features, task_labels):
            x = torch.tensor(x_np, dtype=torch.float32)
            y = torch.tensor(float(y_np), dtype=torch.float32)
            _ = trainer.update(x, y)
            # Évaluation en ligne (optionnel — pour suivi live)
            y_hat, _ = trainer.predict(x)
            n_correct += int((y_hat > 0.5) == bool(y_np))
            total += 1

        print(f"  Train acc (online) : {n_correct / total:.4f}")

        # Évaluation sur toutes les tâches vues (0..task_id)
        task_accs = []
        for eval_id in range(task_id + 1):
            eval_features, eval_labels = cl_stream.get_task(eval_id)
            correct = 0
            for x_np, y_np in zip(eval_features, eval_labels):
                x = torch.tensor(x_np, dtype=torch.float32)
                y_hat, _ = trainer.predict(x)
                correct += int((y_hat > 0.5) == bool(y_np))
            acc = correct / len(eval_labels)
            task_accs.append(acc)
            print(f"  Eval tâche {eval_id + 1} : {acc:.4f}")

        acc_matrix.append(task_accs)

    # Métriques CL finales
    cl_metrics = compute_cl_metrics(acc_matrix)
    print(f"\n[exp_003] AA  = {cl_metrics['acc_final']:.4f}")
    print(f"[exp_003] AF  = {cl_metrics['avg_forgetting']:.4f}")
    print(f"[exp_003] BWT = {cl_metrics['backward_transfer']:.4f}")
```

### 6. Helper : heatmap matrice accuracy

```python
import matplotlib
matplotlib.use("Agg")
import matplotlib.pyplot as plt
import numpy as np


def _save_accuracy_matrix(acc_matrix: list[list[float]], path: Path) -> None:
    """Sauvegarde la heatmap de la matrice d'accuracy CL."""
    n_tasks = len(acc_matrix)
    # Remplir la matrice carrée (NaN pour les tâches futures)
    mat = np.full((n_tasks, n_tasks), np.nan)
    for i, row in enumerate(acc_matrix):
        for j, val in enumerate(row):
            mat[i, j] = val

    fig, ax = plt.subplots(figsize=(6, 5))
    im = ax.imshow(mat, vmin=0, vmax=1, cmap="YlGn")
    ax.set_xlabel("Tâche évaluée")
    ax.set_ylabel("Après tâche d'entraînement")
    ax.set_title("TinyOL — Matrice accuracy CL (exp_003)")
    ax.set_xticks(range(n_tasks))
    ax.set_xticklabels([f"T{i+1}" for i in range(n_tasks)])
    ax.set_yticks(range(n_tasks))
    ax.set_yticklabels([f"T{i+1}" for i in range(n_tasks)])
    for i in range(n_tasks):
        for j in range(i + 1):
            ax.text(j, i, f"{mat[i,j]:.2f}", ha="center", va="center", fontsize=9)
    plt.colorbar(im, ax=ax, label="Accuracy")
    plt.tight_layout()
    plt.savefig(path, dpi=150, bbox_inches="tight")
    plt.close()
    print(f"[exp_003] Accuracy matrix sauvegardée → {path}")
```
