## Préambule : calculer et fixer les feature bounds (Task 1)

Avant d'entraîner HDC, les bornes de quantification doivent être calculées sur Task 1 (Pump) uniquement — comme les stats de normalisation EWC. Cette étape est intégrée au début de `train_hdc.py`.

```python
def compute_feature_bounds_task1(
    train_loader: DataLoader,
) -> dict[str, tuple[float, float]]:
    """
    Calcule min/max de chaque feature sur le train set de Task 1 (Pump).

    À appeler UNE SEULE FOIS (Task 1) et à fixer dans hdc_config.yaml.
    Appliqué à toutes les tâches suivantes sans recalcul (pas de data leakage).

    Returns
    -------
    dict[str, tuple[float, float]]
        {"temperature": (min, max), "pressure": (min, max), ...}
    """
    all_x = []
    for x_batch, _ in train_loader:
        all_x.append(x_batch.numpy())
    X = np.concatenate(all_x, axis=0)  # [N_task1, n_features]

    feature_names = ["temperature", "pressure", "vibration", "humidity"]
    bounds = {}
    for i, name in enumerate(feature_names):
        bounds[name] = (float(X[:, i].min()), float(X[:, i].max()))
    return bounds
```

### 2. Implémenter `train_hdc` (boucle principale)

```python
def evaluate_task_hdc(model: HDCClassifier, val_loader) -> float:
    """Évalue l'accuracy binaire sur un val_loader (DataLoader PyTorch → NumPy)."""
    correct, total = 0, 0
    for x_batch, y_batch in val_loader:
        preds = model.predict(x_batch.numpy())
        correct += int((preds == y_batch.numpy().astype(np.int64)).sum())
        total += len(y_batch)
    return correct / total if total > 0 else 0.0


def train_hdc(
    model: HDCClassifier,
    tasks: list[dict],
    config: dict,
) -> np.ndarray:
    """
    Entraînement HDC Online sur N tâches séquentielles.

    Pas d'optimizer, pas de gradient : accumulation additive de prototypes.

    Parameters
    ----------
    model : HDCClassifier
    tasks : list[dict]
        Sortie de get_cl_dataloaders() — chaque dict contient train_loader + val_loader.
    config : dict
        Configuration hdc_config.yaml.

    Returns
    -------
    acc_matrix : np.ndarray [T, T]
        acc_matrix[i, j] = accuracy sur tâche j après entraînement sur tâche i.
        NaN si j > i (tâche future non encore vue).
    """
    T = len(tasks)
    acc_matrix = np.full((T, T), np.nan)

    for task_idx, task in enumerate(tasks):
        print(f"\n{'='*60}")
        print(f"Tâche {task_idx + 1}/{T} : {task['domain']}")
        print(f"  Train: {task['n_train']} samples | Val: {task['n_val']} samples")

        # Entraînement : parcours unique du train_loader (online, 1 epoch)
        for x_batch, y_batch in task["train_loader"]:
            model.update(x_batch.numpy(), y_batch.numpy())

        # Fin de tâche
        model.on_task_end(task_id=task_idx, dataloader=task["train_loader"])

        # Évaluation sur toutes les tâches vues (backward transfer)
        for eval_idx in range(task_idx + 1):
            acc = evaluate_task_hdc(model, tasks[eval_idx]["val_loader"])
            acc_matrix[task_idx, eval_idx] = acc
            print(f"  Acc tâche {eval_idx + 1} ({tasks[eval_idx]['domain']}): {acc:.4f}")

    return acc_matrix
```

### 3. Implémenter `run_memory_profiling`

```python
def run_memory_profiling(model: HDCClassifier, config: dict) -> dict:
    """
    Profil mémoire du modèle HDC (inférence uniquement — pas de backprop).

    Utilise memory_profiler.full_memory_report() comme pour EWC.
    HDC n'a pas de phase d'update gradient → on profile encode_observation + predict.

    Returns
    -------
    dict
        Rapport mémoire complet (RAM peak, latence, n_params).
    """
    from src.evaluation.memory_profiler import profile_forward_pass

    n_features = config["data"]["n_features"]

    # Wrapper pour rendre HDCClassifier compatible avec profile_forward_pass
    # (qui attend un module torch-like avec forward())
    class HDCWrapper:
        def __call__(self, x):
            return model.predict(x.numpy() if hasattr(x, "numpy") else x)

    # Proxy : profiler l'encodage + prédiction sur 1 sample
    report = profile_forward_pass(
        model=model,
        input_shape=(1, n_features),
        n_runs=100,
        device="cpu",
    )

    # Compléter avec les estimations analytiques
    report["estimated_ram_fp32_bytes"] = model.estimate_ram_bytes("fp32")
    report["estimated_ram_int8_bytes"] = model.estimate_ram_bytes("int8")
    report["within_budget"] = model.check_ram_budget()["within_budget"]
    return report
```

def main() -> None:
    args = parse_args()

def _update_feature_bounds_if_needed(config: dict, task1_loader, config_path: str) -> None:
    """
    Calcule et écrit les feature_bounds dans hdc_config.yaml si elles sont à None.
    Modifie config dict en place.
    """
    bounds_cfg = config.get("feature_bounds", {})
    needs_update = any(
        v is None or (isinstance(v, (list, tuple)) and None in v)
        for v in bounds_cfg.values()
    )
    if not needs_update:
        return

    print("[HDC] Calcul des feature_bounds sur Task 1 (Pump)...")
    bounds = compute_feature_bounds_task1(task1_loader)
    config["feature_bounds"] = {k: list(v) for k, v in bounds.items()}

    # Réécrire le YAML
    with open(config_path, "w") as f:
        yaml.dump(config, f, default_flow_style=False)
    print(f"  feature_bounds écrits dans {config_path}")
    for name, (lo, hi) in bounds.items():
        print(f"    {name}: [{lo:.4f}, {hi:.4f}]")


if __name__ == "__main__":
    main()
